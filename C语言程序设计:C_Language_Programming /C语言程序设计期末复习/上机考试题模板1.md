#define _CRT_SECURE_NO_WARNINGS 1
#include<stdio.h>
#include<string.h>

typedef struct {
	char szName[20];
	double dScore[2];
}STUDENT;

int FailNum(STUDENT* pStu, int nLen);
//这道题给我的困惑主要是忘记了STUDENT *pStu在函数中该如何表示，但是已经解决啦~
int main(void)
{
	int nLen, i;
	STUDENT stu[10];
	STUDENT* pStu;
	pStu = stu;

	scanf("%d", &nLen);


	for (i = 0; i < nLen; i++)
	{
		scanf("%s %lf %lf", stu[i].szName, &stu[i].dScore[0], &stu[i].dScore[1]);
	}

	int sum = FailNum(pStu, nLen);
	printf("%d\n", sum);

	return 0;
}

int FailNum(STUDENT *pStu, int nLen)
{
	int i;
	int count = 0;

	for (i = 0; i < nLen; i++)
	{
		if (pStu[i].dScore[0] < 60 || pStu[i].dScore[1] < 60)
		{
			count++;
		}
	}
	return count;
}
