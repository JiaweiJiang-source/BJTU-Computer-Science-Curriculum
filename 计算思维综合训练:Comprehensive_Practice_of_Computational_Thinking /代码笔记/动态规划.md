## 动态规划

```C
/*
本题是一道关于动态规划的题目，由于每个数的质因数只有2，3，5。因此序列中的每个数都可以表示为2的a次方，3的b次方以及5的c次方。
1, 2, 3, 4, 5, 6, 8, 9, 10, 12, 15, 16, 18, 20, 24, 25, 27, 30, 32, 36, …
我们可以发现，每个元素都有前面的某个元素乘以2、3或者5得到，例如元素8可以由元素4乘2得到。
因此运用三个指针p2,p3和p5来分别表示次数a,b和c，然后对每个位置，去三个指针对应的元素乘以2、3和5中最小的那个数即可。得到的序列则为我们要求的序列。
则时间复杂度为O(N);
*/
#define _CRT_SECURE_NO_WARNINGS 1
#include<stdio.h>

long long int dp[10001] = { 1 }; 

int main() {
    int p2 = 0, p3 = 0, p5 = 0; 
    int i, n;

    for (i = 1; i < 10001; i++) {
        long long int num2 = dp[p2] * 2, num3 = dp[p3] * 3, num5 = dp[p5] * 5;
        long long int next = num2 < num3 ? (num2 < num5 ? num2 : num5) : (num3 < num5 ? num3 : num5); // 取三个数中最小的一个
        if (next == num2) p2++; // 根据最小值移动指针
        if (next == num3) p3++;
        if (next == num5) p5++;
        dp[i] = next;           // 将新的元素加入dp数组中
    }
    while (scanf("%d", &n) != EOF) { // 输入n，直到文件结束
        printf("%lld\n", dp[n - 1]); // 输出第n个元素
    }
    return 0;
}
```
